```python
def quick_sort(nums, l, r):
    if l >= r:
        return
    i, j = l, r
    p = nums[(l + r)//2]
    while i <= j:
        while i <= j and nums[i] < p:
            i += 1
        while i <= j and nums[j] > p:
            j -= 1
        if  i <= j:
            nums[i], nums[j] = nums[j], nums[i]
            i += 1
            j -= 1
    
    quick_sort(nums, l, j)
    quick_sort(nums, i, r)

main()
```

```python
def merge_sort(nums, l, r):
    if l >= r:
        return
    
    mid = (l + r)//2
    
    merge_sort(nums, l, mid)
    merge_sort(nums, mid+1, r)

    tmp = []

        if nums[i] <= nums[j]:
            tmp.append(nums[i])
            i += 1
        else:
            tmp.append(nums[j])
            j += 1
    while i <= mid:
        tmp.append(nums[i])
        i += 1
    while j <= r:
        tmp.append(nums[j])
        j += 1
    
    for i in range(0, len(tmp)):
```

## Binary Search

### Integer Binart Search 

```python
def main():
    des = str(input()).split(" ")
    size, q_num = int(des[0]), int(des[1])
    arr = []
    for char in str(input()).split(" "):
        arr.append(int(char))
    questions = []
    for i in range(0, q_num):
        questions.append(int(input()))
    
    for q in questions:
        find_range(arr, q)

def find_range(nums, target):
    l, r = 0, len(nums)-1
    while l < r:
        mid = (l + r) // 2
        if nums[mid] >= target:
            r = mid
        else:
            l = mid + 1
    
    if nums[l] != target:
        print("-1 -1")
    else:
        print(l, end=" ")
        l, r = 0, len(nums)-1
        while l < r:
            mid = (l + r + 1) // 2
            if nums[mid] <= target:
                l = mid
            else:
                r = mid - 1
        print(l)
    
main()
```

### Float Binary Search

```python
def main():
    num = float(input())
    is_neg = num < 0
    if is_neg:
        num = -num
    l, r = 0, max(num, 1)
    while r - l > 1e-8:
        mid = (l + r) / 2
        if mid ** 3 > num:
            r = mid
        else:
            l = mid
    if is_neg:
        print ("%f" % -l)
        return
    print ("%f" % l)

main()
```
